Is there any difference between these two class definitions
say_hi will be duplicated with every new instance of MyClass1.
Add methods getter/setter methods (i.g. get_name(), set_name(name)) to the classes and change say_hi() text to be "Hi, my name is %name%"!
function MyClass1 () { this.name = "Unknown"; this.say_hi = function() { console.log("Hi, my name is " + this.name); }; this.set_name = function(name) { this.name = name; }; this.get_name = function() { return this.name; }; } function MyClass2 () {} MyClass2.prototype.name = "Unknown"; MyClass2.prototype.say_hi = function() { console.log("Hi, my name is " + this.name); }; MyClass2.prototype.get_name = function() { return this.name; }; MyClass2.prototype.set_name = function(name) { this.name = name; }; var inst1 = new MyClass1(), inst2 = new MyClass2(); inst1.say_hi(); inst2.set_name("Patrick"); inst2.say_hi();
Change class definition to implement the Singleton pattern
var MyClass2 = { name: "Unknown", say_hi: function() { console.log("Hi, my name is " + this.name); }, get_name: function() { return this.name; }, set_name: function(name) { this.name = name; } } inst2 = MyClass2; inst2.set_name("Patrick"); inst2.say_hi();
function MyClass1 () { this.say_hi = function() { console.log('MyClass1 says HI!'); }; } function MyClass2 () {} MyClass2.prototype.say_hi = function() { console.log('MyClass2 says HI!'); }; var inst1 = new MyClass1(), inst2 = new MyClass2(); inst1.say_hi(); inst2.say_hi();
