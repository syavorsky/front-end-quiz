Is there any difference between these two class definitions
The function 'say_hi' in MyClass1 is recreated with every instance of the class, while the function 'say_hi' in MyClass2 is shared between instances of the class.
Add methods getter/setter methods (i.g. get_name(name), set_name()) to the classes and change say_hi() text to be "Hi, my name is %name%"!
function MyClass1 () { this.name = ""; this.say_hi = function() { console.log('Hi, my name is '+this.name+'!'); }; this.get_name = function() { return this.name; } this.set_name = function(name) { this.name = name; } } function MyClass2 () { this.name = "" } MyClass2.prototype.say_hi = function() { console.log('Hi, my name is '+this.name+'!'); }; MyClass2.prototype.get_name = function() { return this.name; } MyClass2.prototype.set_name = function(name) { this.name = name; }
Change class definition to implement the Singleton pattern
var function MyClass1 () { this.name = ""; this.say_hi = function() { console.log('Hi, my name is '+this.name+'!'); }; this.get_name = function() { return this.name; } this.set_name = function(name) { this.name = name; } } var function MyClass2 () { this.name = "" }
function MyClass1 () { this.say_hi = function() { console.log('MyClass1 says HI!'); }; } function MyClass2 () {} MyClass2.prototype.say_hi = function() { console.log('MyClass2 says HI!'); }; var inst1 = new MyClass1(), inst2 = new MyClass2(); inst1.say_hi(); inst2.say_hi();
